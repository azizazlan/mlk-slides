Geth background service

/etc/systemd/system/geth.service : 
            
                [Unit]
                Description=Geth Ethereum client
                After=network.target
                [Service]
                User=YOUR_USERNAME
                Group=YOUR_USERNAME
                ExecStart=/usr/bin/geth --YOUR_GETH_OPTIONS_HERE
                Restart=always
                RestartSec=3
                LimitNOFILE=4096
                [Install]
                WantedBy=multi-user.target
